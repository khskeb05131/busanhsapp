399      private var first_time : Long = 0
400      private var second_time : Long = 0
401      override fun onBackPressed() {
402          second_time = System.currentTimeMillis()
403          if(second_time - first_time < 2000){
404              super.onBackPressed()
405              ActivityCompat.finishAffinity(this)
406              System.runFinalizersOnExit(true)
407              exitProcess(0)
408          }else Toast.makeText(this,"뒤로가기 버튼을 한번 더 눌러 종료하십시오.",Toast.LENGTH_SHORT).show()
409          first_time = System.currentTimeMillis()
410      }
