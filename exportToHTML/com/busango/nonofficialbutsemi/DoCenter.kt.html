16       private val onNavigationItemSelectedListener = BottomNavigationView.OnNavigationItemSelectedListener { item ->
17           when (item.itemId) {
18               R.id.navigation_home -> {
19                   val intent = Intent(this, MainActivity::class.java)
20                   Intent.FLAG_ACTIVITY_CLEAR_TASK
21                   startActivity(intent)
22                   return@OnNavigationItemSelectedListener true
23               }
24               R.id.navigation_dashboard -> {
25                   val intent = Intent(this, DoCenter::class.java)
26                   Intent.FLAG_ACTIVITY_CLEAR_TASK
27                   startActivity(intent)
28                   return@OnNavigationItemSelectedListener true
29               }
30               R.id.navigation_notifications -> {
31                   val intent = Intent(this, SchoolBoard::class.java)
32                   Intent.FLAG_ACTIVITY_CLEAR_TASK
33                   startActivity(intent)
34                   return@OnNavigationItemSelectedListener true
35               }
36           }
37           false
38       }

52       override fun onBackPressed() {
53           second_time = System.currentTimeMillis()
54           if(second_time - first_time < 2000){
55               super.onBackPressed()
56               ActivityCompat.finishAffinity(this)
57               System.runFinalizersOnExit(true)
58               exitProcess(0)
59           }else Toast.makeText(this,"뒤로가기 버튼을 한번 더 눌러 종료하십시오.",Toast.LENGTH_SHORT).show()
